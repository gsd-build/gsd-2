Create packages/mission-control/src/hooks/useChatMode.ts:

The hook needs to receive mode events from the WebSocket. The cleanest approach (per RESEARCH.md recommendation): useChatMode accepts an `onRegisterModeEventHandler` callback, which useSessionManager calls when a mode event arrives. However, to avoid modifying useSessionManager, use a simpler pattern: accept a ref to the raw WebSocket and add an event listener directly.

Actually the cleanest zero-modification approach: expose an `onModeEvent(event: ModeEvent) => void` handler from useChatMode, and in AppShell, pass it to useSessionManager via a new optional `onModeEvent` prop on useSessionManager's message handler.

Given complexity, use the simplest approach that doesn't require modifying useSessionManager: useChatMode accepts the same WebSocket URL and creates its OWN listener on the same URL connection. Use useReconnectingWebSocket directly — same pattern as useSessionManager.

```typescript
import { useState, useCallback, useEffect } from "react";
import { useReconnectingWebSocket } from "./useReconnectingWebSocket";
import { QuestionCard } from "@/components/chat/QuestionCard";
import { DecisionLogDrawer } from "@/components/chat/DecisionLogDrawer";
import type { ModeEvent, QuestionCardPayload, DecisionEntry, ReviewResults } from "@/server/chat-types";

export interface ChatModeState {
  mode: "chat" | "discuss" | "review";
  currentQuestion: QuestionCardPayload | null;
  decisions: DecisionEntry[];
  reviewResults: ReviewResults | null;
  totalQuestions: number;
}

export function useChatMode(wsUrl: string, onChatSend: (msg: string) => void) {
  const [mode, setMode] = useState<"chat" | "discuss" | "review">("chat");
  const [currentQuestion, setCurrentQuestion] = useState<QuestionCardPayload | null>(null);
  const [decisions, setDecisions] = useState<DecisionEntry[]>([]);
  const [reviewResults, setReviewResults] = useState<ReviewResults | null>(null);
  const [totalQuestions, setTotalQuestions] = useState(0);

  const { ws } = useReconnectingWebSocket(wsUrl);

  useEffect(() => {
    if (!ws) return;
    const handler = (ev: MessageEvent) => {
      try {
        const data = JSON.parse(ev.data as string) as ModeEvent;
        if (!data.type || !["discuss_mode_start","question_card","decision_logged","discuss_mode_end","review_mode_start","review_mode_end"].includes(data.type)) return;
        switch (data.type) {
          case "discuss_mode_start":
            setMode("discuss");
            setDecisions([]);
            setTotalQuestions(data.total ?? 0);
            break;
          case "question_card":
            if (data.question) setCurrentQuestion(data.question);
            break;
          case "decision_logged":
            if (data.decision) setDecisions((prev) => [...prev, data.decision!]);
            setCurrentQuestion(null);
            break;
          case "discuss_mode_end":
            setMode("chat");
            setCurrentQuestion(null);
            break;
          case "review_mode_start":
            setMode("review");
            if (data.results) setReviewResults(data.results);
            break;
          case "review_mode_end":
            setMode("chat");
            setReviewResults(null);
            break;
        }
      } catch { /* non-JSON or non-mode event — ignore */ }
    };
    ws.addEventListener("message", handler);
    return () => ws.removeEventListener("message", handler);
  }, [ws]);

  const answerQuestion = useCallback((answer: string) => {
    onChatSend(answer);
  }, [onChatSend]);

  const dismissReview = useCallback(() => {
    setMode("chat");
    setReviewResults(null);
  }, []);

  const handleFix = useCallback((draftMessage: string) => {
    onChatSend(draftMessage);
    dismissReview();
  }, [onChatSend, dismissReview]);

  // Compute overlay React node (only when discuss mode is active and question exists)
  const overlay = mode === "discuss" && currentQuestion ? (
    <div className="flex flex-1">
      <div className="flex-1 flex items-center">
        <QuestionCard question={currentQuestion} onAnswer={answerQuestion} />
      </div>
      <DecisionLogDrawer decisions={decisions} visible={decisions.length > 0} />
    </div>
  ) : undefined;

  return {
    chatModeState: { mode, currentQuestion, decisions, reviewResults, totalQuestions } as ChatModeState,
    overlay,
    reviewResults,
    answerQuestion,
    dismissReview,
    handleFix,
  };
}
```

Note: useReconnectingWebSocket must expose the raw `ws` WebSocket reference. Check if it already does — if it returns `{ ws, status }` already, use that. If it only returns status, add `ws` to its return (minimal change). Check packages/mission-control/src/hooks/useReconnectingWebSocket.ts before implementing.
