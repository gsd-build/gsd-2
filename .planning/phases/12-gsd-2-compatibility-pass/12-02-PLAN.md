GSD 2 file schema (flat root of .gsd/):
  STATE.md            → active milestone, slice, task, auto mode status, cost, tokens
  M{NNN}-ROADMAP.md   → milestone structure, slice checkboxes, risk levels  (NNN from STATE.md active_milestone)
  S{NN}-PLAN.md       → slice task decomposition, must-haves, cost estimate  (NN from STATE.md active_slice)
  T{NN}-SUMMARY.md    → completed task output, YAML frontmatter              (NN from STATE.md active_task)
  DECISIONS.md        → architectural decision register
  preferences.md      → model config, budget ceiling, skill_discovery setting (YAML frontmatter)
  PROJECT.md          → living project description
  M{NNN}-CONTEXT.md   → user decisions for active milestone

GSD2State shape (from RESEARCH.md Pattern 2):
```typescript
export interface GSD2State {
  projectState: GSD2ProjectState;
  roadmap: GSD2RoadmapState | null;
  activePlan: GSD2SlicePlan | null;
  activeTask: GSD2TaskSummary | null;
  decisions: string | null;
  preferences: GSD2Preferences | null;
  project: string | null;
  milestoneContext: string | null;
  needsMigration: boolean;
}

export interface GSD2ProjectState {
  gsd_state_version: string;
  milestone: string;
  milestone_name: string;
  status: string;
  active_milestone: string;   // "M001", "M002", etc.
  active_slice: string;       // "S01", "S02", etc.
  active_task: string;        // "T01", "T02", etc.
  auto_mode: boolean;
  cost: number;
  tokens: number;
  last_updated: string;
  last_activity?: string;
}

export interface GSD2Preferences {
  research_model?: string;
  planning_model?: string;
  execution_model?: string;
  completion_model?: string;
  budget_ceiling?: number;
  skill_discovery?: "auto" | "suggest" | "off";
}
```
