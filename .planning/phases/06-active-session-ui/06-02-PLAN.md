Spinner pattern from research (Pattern 4):
```typescript
const [pendingAction, setPendingAction] = useState<string | null>(null)
const pendingTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null)

const handlePrimaryAction = () => {
  if (!workflowAction.primary) return
  setPendingAction(workflowAction.primary.command)
  pendingTimerRef.current = setTimeout(() => {
    setPendingAction(null)  // Revert after 3s
  }, 3000)
  handleWorkflowAction(workflowAction.primary.command)
}

// Clear spinner when store confirms state change
useEffect(() => {
  if (pendingAction === null) return
  setPendingAction(null)
  if (pendingTimerRef.current) clearTimeout(pendingTimerRef.current)
}, [isAutoActive, autoPaused])
```
