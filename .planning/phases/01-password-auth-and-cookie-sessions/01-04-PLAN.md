    ```typescript
    import { readFile, writeFile, mkdir } from 'node:fs/promises'
    import { join } from 'node:path'
    import { hashPassword, rotateSessionSecret } from './web-session-auth.js'
    import { appRoot, webPreferencesPath } from '../app-paths.js'

    export async function setPassword(password: string, gsdDir?: string): Promise<void> {
      const dir = gsdDir ?? appRoot
      const prefsPath = gsdDir ? join(gsdDir, 'web-preferences.json') : webPreferencesPath

      // Hash the password
      const passwordHash = await hashPassword(password)

      // Read existing preferences (read-modify-write to preserve other keys)
      let prefs: Record<string, unknown> = {}
      try {
        const raw = await readFile(prefsPath, 'utf-8')
        prefs = JSON.parse(raw)
      } catch {
        // File doesn't exist yet — start fresh
      }

      // Write updated preferences
      prefs.passwordHash = passwordHash
      await mkdir(dir, { recursive: true })
      await writeFile(prefsPath, JSON.stringify(prefs, null, 2), 'utf-8')

      // Rotate session secret to invalidate all existing sessions (AUTH-07)
      await rotateSessionSecret(dir)
    }
    ```

    The function is intentionally simple — it's glue between the crypto primitives (Plan 01) and the preferences file. The real crypto work happens in `hashPassword` and `rotateSessionSecret`.
