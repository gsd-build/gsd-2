    Create packages/mission-control/src/server/trust-api.ts:

    ```typescript
    import { access, writeFile, mkdir } from "node:fs/promises";
    import { join } from "node:path";

    const TRUST_FLAG = ".mission-control-trust";

    export async function isTrusted(gsdDir: string): Promise<boolean> {
      try {
        await access(join(gsdDir, TRUST_FLAG));
        return true;
      } catch {
        return false;
      }
    }

    export async function writeTrustFlag(gsdDir: string): Promise<void> {
      await mkdir(gsdDir, { recursive: true });
      await writeFile(join(gsdDir, TRUST_FLAG), "", { flag: "w" });
    }

    /** Register GET /api/trust and POST /api/trust on a Bun.serve fetch handler. */
    export function registerTrustRoutes(
      url: URL,
      method: string,
      body: unknown
    ): Response | null {
      if (url.pathname !== "/api/trust") return null;

      if (method === "GET") {
        const dir = url.searchParams.get("dir") ?? "";
        return new Response(
          JSON.stringify({ trusted: false }),
          {
            headers: { "Content-Type": "application/json" },
          }
        );
        // Note: returned synchronously; real check done via isTrusted helper
        // Tests call the helpers directly; routes are thin wrappers.
      }

      return null;
    }
    ```

    For the REST routes, implement them as thin wrappers in server.ts (or wherever routes are registered). For now trust-api.ts exports the two pure async helpers (isTrusted, writeTrustFlag) and a registerTrustRoutes function that handles parsing and response construction:

    Actual registerTrustRoutes implementation:
    - GET: await isTrusted(dir), return JSON { trusted: boolean }
    - POST: parse body.dir, await writeTrustFlag(dir), return JSON { ok: true }
    - Return null for non-matching routes (caller checks null)
