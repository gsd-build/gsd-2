Add exported pure function `scanForDevServers`:
```typescript
const CANDIDATE_PORTS = [3000, 4173, 5173, 8080, 8000];

export async function scanForDevServers(): Promise<DetectedServer[]> {
  const results = await Promise.allSettled(
    CANDIDATE_PORTS.map(async (port) => {
      const controller = new AbortController();
      const timer = setTimeout(() => controller.abort(), 800);
      try {
        await fetch(`http://localhost:${port}/`, {
          signal: controller.signal,
          mode: "no-cors",
        });
        clearTimeout(timer);
        // Port heuristic for type classification:
        const type = [5173, 4173, 3000].includes(port) ? "frontend" as const
          : [8080, 8000].includes(port) ? "backend" as const
          : "unknown" as const;
        const label = port === 5173 ? "Vite" : port === 4173 ? "Vite Preview" : port === 3000 ? "Dev" : port === 8080 ? "API" : port === 8000 ? "API" : `Port ${port}`;
        return { port, type, label: `${label} (:${port})` };
      } catch {
        clearTimeout(timer);
        throw new Error(`Port ${port} not responding`);
      }
    })
  );
  return results
    .filter((r): r is PromiseFulfilledResult<DetectedServer> => r.status === "fulfilled")
    .map((r) => r.value);
}
```
